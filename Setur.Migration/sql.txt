CREATE SCHEMA IF NOT EXISTS dbo;
DROP TABLE IF EXISTS dbo.DIRECTORY;
Create table dbo.DIRECTORY(
ID SERIAL,
UUID CHAR(32),
NAME CHAR(50),
SURNAME CHAR(100),
COMPANY CHAR(200)
);
DROP TABLE IF EXISTS dbo.CONTACT_INFORMATION;
CREATE TABLE dbo.CONTACT_INFORMATION(
ID SERIAL,
DIRECTORY_UUID VARCHAR(32),
TYPE SMALLINT,
CONTENT CHAR(500)
);
DROP TABLE IF EXISTS dbo.REPORT;
CREATE TABLE dbo.REPORT(
ID SERIAL,
UUID CHAR(32),
CREATE_DATE TIME,
STATUS SMALLINT
);
DROP TABLE IF EXISTS dbo.REPORT_DETAIL;
CREATE TABLE dbo.REPORT_DETAIL(
ID SERIAL,
REPORT_UUID CHAR(32),
LOCATION CHAR(500),
NUMBER_OF_PEOPLE INT,
NUMBER_OF_PHONE_NUMBER INT
);
